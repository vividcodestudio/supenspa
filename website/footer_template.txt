    <script>
        // Carousel — no dots, just smooth auto-rotation
        const slides = document.querySelectorAll('.carousel-slide');
        const progressBar = document.getElementById('heroProgressBar');
        let current = 0;
        const INTERVAL = 7000;

        function resetProgress() {
            if (!progressBar) return;
            progressBar.style.transition = 'none';
            progressBar.style.width = '0%';
            // Force reflow then animate
            progressBar.getBoundingClientRect();
            progressBar.style.transition = 'width ' + INTERVAL + 'ms linear';
            progressBar.style.width = '100%';
        }

        function goTo(n) {
            if (slides.length === 0) return;
            const outgoing = slides[current];
            outgoing.classList.remove('active');
            outgoing.classList.add('leaving');

            current = (n + slides.length) % slides.length;
            slides[current].classList.add('active');

            // Clean up .leaving after crossfade completes (2.4s)
            setTimeout(() => outgoing.classList.remove('leaving'), 2400);

            resetProgress();
        }

        if (slides.length > 1) {
            resetProgress();
            setInterval(() => goTo(current + 1), INTERVAL);
        }

        if (slides.length > 0) slides[0].classList.add('active');

        // Scroll Reveal — reliable version
        const revealObserver = new IntersectionObserver(entries => {
            entries.forEach(e => {
                if (e.isIntersecting) {
                    e.target.classList.add('visible');
                    revealObserver.unobserve(e.target);
                }
            });
        }, { threshold: 0, rootMargin: '0px 0px -60px 0px' });

        const revealEls = document.querySelectorAll('.scroll-reveal');
        revealEls.forEach(el => revealObserver.observe(el));

        // Hard fallback: force-show any still-hidden elements after 1.8s
        setTimeout(() => {
            revealEls.forEach(el => el.classList.add('visible'));
        }, 1800);

        // Mobile Menu
        function toggleMenu() {
            const menu = document.getElementById('nav-menu');
            const hbg = document.getElementById('hamburger');
            menu.classList.toggle('active');
            hbg.classList.toggle('open');
        }
    </script>
